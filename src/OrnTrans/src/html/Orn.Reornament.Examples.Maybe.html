-- Paper: Example 4.31

iMaybe : Bool → Set
iMaybe b = μ ⌈ MaybeO A ⌉D (tt , b)

inothing : iMaybe false
inothing = ⟨ tt ⟩ 

ijust : A → iMaybe true
ijust a = ⟨ a , tt ⟩
