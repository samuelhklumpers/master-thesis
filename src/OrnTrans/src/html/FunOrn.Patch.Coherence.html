-- Paper: Definition 5.23
⊢Coherence : {T : Type } → (T⁺ : FunctionOrn T)(f : ⟦ T ⟧Type)
             (p : Patch f T⁺) → ⟦ T⁺ ⟧Coherence f (patch T⁺ f p)
⊢Coherence (μ⁺ o [ inv i⁺ ]→ T⁺) f f⁺ = 
  λ x → ⊢Coherence T⁺ 
                   (f (forget o x)) 
                   (f⁺ (forget o x)
                       (make o x))
⊢Coherence (μ⁺ o [ inv i⁺ ]× T⁺) (x , xs) (x⁺ , xs⁺) = 
  coherentOrn o x⁺ , ⊢Coherence T⁺ xs xs⁺
⊢Coherence `⊤ tt tt = tt
