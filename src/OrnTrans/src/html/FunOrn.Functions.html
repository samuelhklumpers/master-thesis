data Type : Set₁ where
  μ_[_]→_ : {I : Set}(D : func  I I)(i : I)(T : Type) → Type 
  μ_[_]×_ : {I : Set}(D : func  I I)(i : I)(T : Type) → Type 
  `⊤ : Type 

⟦_⟧Type : Type → Set 
⟦ μ D [ i ]→ T ⟧Type = μ D i → ⟦ T ⟧Type
⟦ μ D [ i ]× T ⟧Type = μ D i × ⟦ T ⟧Type
⟦ `⊤ ⟧Type = ⊤
