-- Paper: Example 3.6

ListD : Set → func ⊤ ⊤
ListD A = func.mk λ _ →
       `σ 2 
          (λ { zero → `1 
             ; (suc zero) → `Σ A λ _ → `var tt
             ; (suc (suc ())) })

List : Set → Set
List A = μ (ListD A) tt

nil : ∀{A} → List A
nil = ⟨ zero , tt ⟩

cons : ∀{A} → A → List A → List A
cons a xs = ⟨ suc zero , a , xs ⟩
